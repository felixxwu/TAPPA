# PS1-Style Drivable Car Prototype Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A single Godot scene with a checker-textured floor and a keyboard-drivable VehicleBody3D car, rendered in PS1 style (320×240, unshaded, vertex wobble, affine mapping, color quantization + dithering).

**Architecture:** One main scene built as a text `.tscn`. Two custom shaders (a spatial model shader and a canvas post-process shader) provide the entire PS1 look — no lights exist in the scene. Two small GDScripts drive the car and the chase camera.

**Tech Stack:** Godot 4.6, GL Compatibility renderer, Jolt physics, GDScript, Godot shading language.

**Note:** This project is intentionally not under git (user declined `git init`). All "Commit" steps from the standard template are replaced by verification steps. There is no test framework in this project; per the spec, verification is manual via the running scene plus `--headless --check-only` style launches to catch script/shader parse errors.

```bash
import struct, zlib
size, cell = 64, 8
rows = bytearray()
for y in range(size):
    rows.append(0)  # PNG filter byte per scanline
    for x in range(size):
        c = (170, 170, 170) if ((x // cell + y // cell) % 2 == 0) else (90, 90, 90)
        rows.extend(c)
def chunk(tag, data):
    return struct.pack('>I', len(data)) + tag + data + struct.pack('>I', zlib.crc32(tag + data))
png = b'\x89PNG\r\n\x1a\n'
png += chunk(b'IHDR', struct.pack('>IIBBBBB', size, size, 8, 2, 0, 0, 0))
png += chunk(b'IDAT', zlib.compress(bytes(rows)))
png += chunk(b'IEND', b'')
open('textures/checker.png', 'wb').write(png)
print('wrote textures/checker.png')
EOF
```

### Task 2: PS1 model shader (vertex snap + affine mapping)

**Files:**
- Create: `shaders/ps1_models.gdshader`

- [ ] **Step 1: Write the shader**

Create `shaders/ps1_models.gdshader`:

```glsl
shader_type spatial;
render_mode unshaded, specular_disabled;

uniform sampler2D albedo_texture : source_color, filter_nearest, hint_default_white;
uniform vec4 albedo_color : source_color = vec4(1.0);
uniform vec2 texture_tile = vec2(1.0, 1.0);
uniform bool snap_enabled = true;
uniform float snap_resolution = 240.0;
uniform float affine_weight : hint_range(0.0, 1.0) = 1.0;

varying float clip_w;
varying vec2 uv_affine;
varying vec2 uv_perspective;

void vertex() {
	vec4 clip_pos = PROJECTION_MATRIX * MODELVIEW_MATRIX * vec4(VERTEX, 1.0);
	if (snap_enabled) {
		vec2 ndc = clip_pos.xy / clip_pos.w;
		ndc = round(ndc * snap_resolution) / snap_resolution;
		clip_pos.xy = ndc * clip_pos.w;
	}
	POSITION = clip_pos;
	clip_w = clip_pos.w;
	uv_perspective = UV * texture_tile;
	// Varyings interpolate perspective-correct by default; multiplying by w
	// here and dividing by interpolated w in fragment() yields screen-linear
	// (affine) interpolation, the PS1 artifact.
	uv_affine = uv_perspective * clip_w;
}

void fragment() {
	vec2 uv = mix(uv_perspective, uv_affine / clip_w, affine_weight);
	ALBEDO = texture(albedo_texture, uv).rgb * albedo_color.rgb;
}
```

- [ ] **Step 2: Verify**

Shader parse errors surface when the editor first loads the scene (Task 5, Step 2). No standalone shader compiler exists; re-read the file checking: `POSITION` is written exactly once, all three varyings are assigned in `vertex()` and read in `fragment()`.

### Task 3: PS1 post-process shader (quantization + dithering)

**Files:**
- Create: `shaders/ps1_post_process.gdshader`

- [ ] **Step 1: Write the shader**

Create `shaders/ps1_post_process.gdshader`:

```glsl
shader_type canvas_item;

uniform sampler2D screen_texture : hint_screen_texture, repeat_disable, filter_nearest;
uniform vec2 virtual_resolution = vec2(320.0, 240.0);

// The PS1's actual 4x4 ordered dither matrix. Values are added to the
// 0-255 color before truncation to 5 bits per channel.
const mat4 DITHER = mat4(
	vec4(-4.0, 2.0, -3.0, 3.0),
	vec4(0.0, -2.0, 1.0, -1.0),
	vec4(-3.0, 3.0, -4.0, 2.0),
	vec4(1.0, -1.0, 0.0, -2.0)
);

void fragment() {
	vec3 color = texture(screen_texture, SCREEN_UV).rgb;
	vec2 virtual_pixel = floor(SCREEN_UV * virtual_resolution);
	int x = int(mod(virtual_pixel.x, 4.0));
	int y = int(mod(virtual_pixel.y, 4.0));
	float dither = DITHER[x][y] / 255.0;
	vec3 quantized = floor((color + dither) * 255.0 / 8.0) / 31.0;
	COLOR = vec4(quantized, 1.0);
}
```

Note: `mat4` is column-major; `DITHER[x][y]` indexes column x, row y, matching the matrix layout above (each `vec4` is one column).

- [ ] **Step 2: Verify**

Same as Task 2 — parse check happens at first scene load (Task 5). Re-read: `screen_texture` uses `hint_screen_texture`, `COLOR` alpha is forced to 1.0, quantization divides by 31 (5-bit), not 32.

### Task 4: Car and camera scripts

**Files:**
- Create: `scripts/car.gd`
- Create: `scripts/chase_camera.gd`

- [ ] **Step 1: Write car.gd**

Create `scripts/car.gd`:

```gdscript
extends VehicleBody3D

const MAX_ENGINE_FORCE := 120.0
const MAX_BRAKE := 3.0
const STEER_LIMIT := 0.5
const STEER_SPEED := 5.0

var _start_transform: Transform3D


func _ready() -> void:
	_start_transform = global_transform


func _physics_process(delta: float) -> void:
	var throttle := Input.get_axis("brake_reverse", "accelerate")
	engine_force = throttle * MAX_ENGINE_FORCE
	brake = MAX_BRAKE if (throttle < 0.0 and linear_velocity.dot(-global_transform.basis.z) > 1.0) else 0.0

	var steer_target := Input.get_axis("steer_right", "steer_left") * STEER_LIMIT
	steering = move_toward(steering, steer_target, STEER_SPEED * delta)

	if Input.is_action_just_pressed("reset_car"):
		_reset()


func _reset() -> void:
	global_transform = _start_transform
	linear_velocity = Vector3.ZERO
	angular_velocity = Vector3.ZERO
```

Behavior notes: holding S while moving forward applies brake; once nearly stopped (or already reversing), S becomes reverse via negative `engine_force`. VehicleBody3D forward is -Z.

- [ ] **Step 2: Write chase_camera.gd**

Create `scripts/chase_camera.gd`:

```gdscript
extends Camera3D

@export var target: Node3D
@export var distance := 6.0
@export var height := 3.0
@export var smoothing := 5.0


func _physics_process(delta: float) -> void:
	if target == null:
		return
	var back: Vector3 = target.global_transform.basis.z
	back.y = 0.0
	back = back.normalized()
	var desired := target.global_position + back * distance + Vector3.UP * height
	global_position = global_position.lerp(desired, 1.0 - exp(-smoothing * delta))
	look_at(target.global_position, Vector3.UP)
```

The `1.0 - exp(-smoothing * delta)` form makes the lerp framerate-independent.
